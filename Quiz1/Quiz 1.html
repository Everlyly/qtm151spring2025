list_emissions=[]
name=["data_quiz/china_co2_emissions.csv","data_quiz/india_co2_emissions.csv"]

for x in name:
    data=pd.read_csv(x)
    mean=data["total_emissions"].mean()
    list_emissions.append(mean)
